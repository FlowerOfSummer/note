interface Cat {
  type: 'cat'
  breeds: 'Abyssinian' | 'Shorthair' | 'Curl' | 'Bengal'
}

interface Dog {
  type: 'dog'
  breeds: 'Hound' | 'Brittany' | 'Bulldog' | 'Boxer'
  color: 'brown' | 'white' | 'black'
}

// type LookUp
type LookUp<type, key> = (type, key) => type.type === key
// -> Dog
type MyDogType = LookUp<Cat | Dog, 'dog'>
