let map = {}
function flat(arr) {
    return arr.reduce((pre, cur) => {
        if(Array.isArray(cur)) {
            pre = [...pre, ...flat(cur)]
        } else {
            if(!map[cur])  {
               pre.push(cur) 
                map[cur] = 1
            }
        }
        return pre;
    }, [])
}
