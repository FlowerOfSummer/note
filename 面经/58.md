function sortArr(arr) {
  let i = arr.length;
  while (i) {
    let j = parseInt(Math.random() * i--);
    [arr[i], arr[j]] = [arr[j], arr[i]]
  }
  return arr;
}
function getRandom(a) {
    return Math.random()*(a - 0.01) + 0.01;
}
